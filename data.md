```elm {l=hidden}
classesPerSpell : Table
classesPerSpell =
    spellTable
      |> gather "Class" "bool"[ ( "Cleric", "cleric" ),( "Bard", "bard" ),( "Druid", "druid" ),( "Paladin", "paladin" ),( "Ranger", "ranger" ),( "Sorcerer", "sorcerer" ),( "Warlock", "warlock" ),( "Wizard", "wizard" ) ]
      |> filterRows "bool" (\v -> v =="TRUE")

```

```elm {l=hidden}
rangePropertySpells : Table
rangePropertySpells =
    classesPerSpell
      |> gather "rangeType" "rangeBool"[("Touch","Touch"),("Aura","Aura")]
```

```elm {l=hidden}
selfSpells : Table
selfSpells =
    rangePropertySpells
      |> filterRows "Range" (\v -> v == "0")
      |> filterRows "rangeType" (\v -> v == "Touch")


```

```elm {l=hidden}
aura : Table
aura =
    rangePropertySpells
      |> filterRows "rangeBool" (\v -> v == "TRUE")
      |> filterRows "rangeType" (\v -> v == "Aura")
```
